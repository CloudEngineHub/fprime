62 F32 SignalGen::generateSample(U32 ticks) {
63 F32 val = 0.0f;
64 if (this->skipOne) {
65 return val;
66 }
67 // Samples per period
68 F32 samplesPerPeriod = static_cast<F32>(this->sampleFrequency) / static_cast<F32>(this->signalFrequency);
69 U32 halfSamplesPerPeriod = samplesPerPeriod / 2;
70 /* Signals courtesy of the open source Aquila DSP Library */
71 switch (this->sigType.e) {
72 case SignalType::TRIANGLE:
73 {
74 F32 m = this->signalAmplitude / static_cast<F32>(halfSamplesPerPeriod);
75 val = m * static_cast<F32>(ticks % halfSamplesPerPeriod);
76 break;
77 }
78 case SignalType::SINE:
79 {
80 F32 normalizedFrequency = 1.0f / samplesPerPeriod;
81 val = this->signalAmplitude * std::sin((2.0 * M_PI * normalizedFrequency *
82 static_cast<F32>(ticks)) + (this->signalPhase * 2.0 * M_PI));
83 break;
84 }
85 case SignalType::SQUARE:
86 {
87 val = this->signalAmplitude * ((ticks % static_cast<U32>(samplesPerPeriod) < halfSamplesPerPeriod) ? 1.0f : -1.0f);
88 break;
89 }
90 case SignalType::NOISE:
91 {
92 val = this->signalAmplitude * (std::rand() / static_cast<double>(RAND_MAX));
93 break;
94 }
95 default:
96 FW_ASSERT(0); // Should never happen
97 }
98 return val;
99 }

101 void SignalGen::schedIn_handler(
102 NATIVE_INT_TYPE portNum,
103 U32 context
104 )
105 {
106 F32 value = 0.0f;
107 // This is a queued component, so it must intentionally run the dispatch of commands and queue processing on this
108 // synchronous scheduled call
109 this->doDispatch();
110
111 // This short-circuits when the signal generator is not running
112 if (not this->running) {
113 return;
114 }
115 // Allows for skipping a single reading of the signal
116 if (not this->skipOne) {
117 value = this->generateSample(this->ticks);
118 }
119 this->skipOne = false;
120
121 // Build our new types
122 SignalPair pair = SignalPair(this->ticks, value);
123
124 // Shift and assign our array types
125 for (U32 i = 1; i < this->sigHistory.SIZE; i++) {
126 this->sigHistory[i - 1] = this->sigHistory[i];
127 this->sigPairHistory[i - 1] = this->sigPairHistory[i];
128 }
129 this->sigHistory[this->sigHistory.SIZE - 1] = value;
130 this->sigPairHistory[this->sigPairHistory.SIZE - 1] = pair;
131
132 // Composite structure
133 SignalInfo sigInfo(this->sigType, this->sigHistory, this->sigPairHistory);
134
135 // Write all signals
136 this->tlmWrite_Type(this->sigType);
137 this->tlmWrite_Output(value);
138 this->tlmWrite_PairOutput(pair);
139 this->tlmWrite_History(this->sigHistory);
140 this->tlmWrite_PairHistory(this->sigPairHistory);
141 this->tlmWrite_Info(sigInfo);
142
143 // if a Data product is being generated, store a record
144 if (this->m_dpInProgress) {
145 Fw::SerializeStatus stat = this->m_dpContainer.serializeRecord_DataRecord(sigInfo);
146 this->m_currDp++;
147 this->m_dpBytes += SignalInfo::SERIALIZED_SIZE;
148 // check for full data product
149 if (Fw::SerializeStatus::FW_SERIALIZE_NO_ROOM_LEFT == stat) {
150 this->log_WARNING_LO_SignalGen_DpRecordFull(this->m_currDp,this->m_dpBytes);
151 this->cleanupAndSendDp();
152 } else if (this->m_currDp == this->m_numDps) { // if we reached the target number of DPs
153 this->log_ACTIVITY_LO_SignalGen_DpComplete(this->m_numDps,this->m_dpBytes);
154 this->cleanupAndSendDp();
155 }
156
157 this->tlmWrite_DpBytes(this->m_dpBytes);
158 this->tlmWrite_DpRecords(this->m_currDp);
159 }
160
161 this->ticks += 1;
162 }

164 void SignalGen::SignalGen_Settings_cmdHandler(
165 FwOpcodeType opCode,
166 U32 cmdSeq,
167 U32 Frequency,
168 F32 Amplitude,
169 F32 Phase,
170 Ref::SignalType SigType
171 )
172 {
173 this->signalFrequency = Frequency;
174 this->signalAmplitude = Amplitude;
175 this->signalPhase = Phase;
176 this->sigType = SigType;
177
178 // When the settings change, reset the history values
179 for (U32 i = 0; i < SignalSet::SIZE; i++) {
180 this->sigHistory[i] = 0.0f;
181 }
182 for (U32 i = 0; i < SignalPairSet::SIZE; i++) {
183 this->sigPairHistory[i].settime(0.0f);
184 this->sigPairHistory[i].setvalue(0.0f);
185 }
186 this->log_ACTIVITY_LO_SignalGen_SettingsChanged(this->signalFrequency, this->signalAmplitude, this->signalPhase, this->sigType);
187 this->tlmWrite_Type(SigType);
188 this->cmdResponse_out(opCode, cmdSeq, Fw::CmdResponse::OK);
189 }
