56 void RpiDemoComponentImpl::preamble() {
57 // check initial state parameter
58 Fw::ParamValid valid;
59 RpiDemo_LedState initState = paramGet_RD_PrmLedInitState(valid);
60 // check status
61 switch (valid.e) {
62 // if default or valid, use stored value
63 case Fw::ParamValid::DEFAULT:
64 case Fw::ParamValid::VALID:
65 this->m_ledOn = (RpiDemo_LedState::BLINKING == initState.e);

67 this->m_ledOn ?

69 );
70 break;

81 void RpiDemoComponentImpl ::
82 Run_handler(
83 const NATIVE_INT_TYPE portNum,
84 U32 context
85 )
86 {
87 // check which rate group call it is
88 switch (context) {
89 case RG_CONTEXT_1Hz:
90 // write telemetry channels
91 this->tlmWrite_RD_LastMsg(this->m_lastUartMsg);
92 this->tlmWrite_RD_UartRecvBytes(this->m_uartReadBytes);
93 this->tlmWrite_RD_UartSentBytes(this->m_uartWriteBytes);
94 this->tlmWrite_RD_SpiBytes(this->m_spiBytes);
95 this->tlmWrite_RD_1HzTicks(this->m_1HzTicks);
96 this->tlmWrite_RD_10HzTicks(this->m_10HzTicks);
97 this->m_1HzTicks++;
98 break;
99 case RG_CONTEXT_10Hz:
100 // Toggle LED value
101 if ( (this->m_10HzTicks++%this->m_ledDivider == 0) and this->m_ledOn) {
102 this->GpioWrite_out(2, this->m_currLedVal);
103 this->m_currLedVal = (this->m_currLedVal == Fw::Logic::HIGH) ?
104 Fw::Logic::LOW : Fw::Logic::HIGH;
105 }
106 break;
107 default:
108 FW_ASSERT(0, context);
109 break; // for the code checkers
110 }

114 void RpiDemoComponentImpl ::
115 UartRead_handler(
116 const NATIVE_INT_TYPE portNum,
117 Fw::Buffer &serBuffer,
118 const Drv::RecvStatus &status
119 )
120 {
121 if (Drv::RecvStatus::RECV_OK == status.e) {
122 // convert incoming data to string. If it is not printable, set character to '*'
123 char uMsg[serBuffer.getSize() + 1];
124 char *bPtr = reinterpret_cast<char *>(serBuffer.getData());
125
126 for (NATIVE_UINT_TYPE byte = 0; byte < serBuffer.getSize(); byte++) {
127 uMsg[byte] = isalpha(bPtr[byte]) ? bPtr[byte] : '*';
128 }
129 uMsg[sizeof(uMsg) - 1] = 0;
130
131 Fw::LogStringArg evrMsg(uMsg);
132 this->log_ACTIVITY_HI_RD_UartMsgIn(evrMsg);
133 this->m_lastUartMsg = uMsg;
134 this->m_uartReadBytes += serBuffer.getSize();
135 }
136 // return buffer to buffer manager
137 this->UartBuffers_out(0, serBuffer);
138 }

224 void RpiDemoComponentImpl ::
225 RD_SendSpi_cmdHandler(
226 const FwOpcodeType opCode,
227 const U32 cmdSeq,
228 const Fw::CmdStringArg& data
229 )
230 {
231 // copy data from string to output buffer
232 char inBuf[data.length()+1];
233 Fw::Buffer in;
234 in.setData(reinterpret_cast<U8*>(inBuf));
235 in.setSize(sizeof(inBuf));
236
237 Fw::Buffer out;
238 out.setData(reinterpret_cast<U8*>(const_cast<char*>(data.toChar())));
239 out.setSize(data.length());
240 this->SpiReadWrite_out(0, out, in);
241 for (NATIVE_UINT_TYPE byte = 0; byte < sizeof(inBuf); byte++) {
242 inBuf[byte] = isalpha(inBuf[byte])?inBuf[byte]:'*';
243 }
244 inBuf[sizeof(inBuf)-1] = 0;
245 // write reply to event
246 Fw::LogStringArg arg = inBuf;
247 this->log_ACTIVITY_HI_RD_SpiMsgIn(arg);
248 this->m_spiBytes += data.length();
249
250 this->cmdResponse_out(opCode, cmdSeq, Fw::CmdResponse::OK);
251 }
