32/*
33 * random.c:
34 *
35 * An improved random number generation package. In addition to the standard
36 * rand()/srand() like interface, this package also has a special state info
37 * interface. The initstate() routine is called with a seed, an array of
38 * bytes, and a count of how many bytes are being passed in; this array is
39 * then initialized to contain information for random number generation with
40 * that much state information. Good sizes for the amount of state
41 * information are 32, 64, 128, and 256 bytes. The state can be switched by
42 * calling the setstate() routine with the same array as was initialized
43 * with initstate(). By default, the package runs with 128 bytes of state
44 * information and generates far better random numbers than a linear
45 * congruential generator. If the amount of state information is less than
46 * 32 bytes, a simple linear congruential R.N.G. is used.
47 *
48 * Internally, the state information is treated as an array of uint32_t's; the
49 * zeroeth element of the array is the type of R.N.G. being used (small
50 * integer); the remainder of the array is the state information for the
51 * R.N.G. Thus, 32 bytes of state information will give 7 ints worth of
52 * state information, which will allow a degree seven polynomial. (Note:
53 * the zeroeth word of state information also has some other information
54 * stored in it -- see setstate() for details).
55 *
56 * The random number generation technique is a linear feedback shift register
57 * approach, employing trinomials (since there are fewer terms to sum up that
58 * way). In this approach, the least significant bit of all the numbers in
59 * the state table will act as a linear feedback shift register, and will
60 * have period 2^deg - 1 (where deg is the degree of the polynomial being
61 * used, assuming that the polynomial is irreducible and primitive). The
62 * higher order bits will have longer periods, since their values are also
63 * influenced by pseudo-random carries out of the lower bits. The total
64 * period of the generator is approximately deg*(2**deg - 1); thus doubling
65 * the amount of state information has a vast influence on the period of the
66 * generator. Note: the deg*(2**deg - 1) is an approximation only good for
67 * large deg, when the period of the shift is the dominant factor.
68 * With deg equal to seven, the period is actually much longer than the
69 * 7*(2**7 - 1) predicted by this formula.
70 *
71 * Modified 28 December 1994 by Jacob S. Rosenberg.
72 * The following changes have been made:
73 * All references to the type u_int have been changed to unsigned long.
74 * All references to type int have been changed to type long. Other
75 * cleanups have been made as well. A warning for both initstate and
76 * setstate has been inserted to the effect that on Sparc platforms
77 * the 'arg_state' variable must be forced to begin on word boundaries.
78 * This can be easily done by casting a long integer array to char *.
79 * The overall logic has been left STRICTLY alone. This software was
80 * tested on both a VAX and Sun SparcStation with exactly the same
81 * results. The new version and the original give IDENTICAL results.
82 * The new version is somewhat faster than the original. As the
83 * documentation says: "By default, the package runs with 128 bytes of
84 * state information and generates far better random numbers than a linear
85 * congruential generator. If the amount of state information is less than
86 * 32 bytes, a simple linear congruential R.N.G. is used." For a buffer of
87 * 128 bytes, this new version runs about 19 percent faster and for a 16
88 * byte buffer it is about 5 percent faster.
89 */

91/*
92 * For each of the currently supported random number generators, we have a
93 * break value on the amount of state information (you need at least this
94 * many bytes of state info to support this random number generator), a degree
95 * for the polynomial (actually a trinomial) that the R.N.G. is based on, and
96 * the separation between the two lower order coefficients of the trinomial.
97 */
98#define TYPE_0 0 /* linear congruential */
99#define BREAK_0 8
100#define DEG_0 0
101#define SEP_0 0
102
103#define TYPE_1 1 /* x**7 + x**3 + 1 */
104#define BREAK_1 32
105#define DEG_1 7
106#define SEP_1 3
107
108#define TYPE_2 2 /* x**15 + x + 1 */
109#define BREAK_2 64
110#define DEG_2 15
111#define SEP_2 1
112
113#define TYPE_3 3 /* x**31 + x**3 + 1 */
114#define BREAK_3 128
115#define DEG_3 31
116#define SEP_3 3
117
118#define TYPE_4 4 /* x**63 + x + 1 */
119#define BREAK_4 256
120#define DEG_4 63
121#define SEP_4 1

123/*
124 * Array versions of the above information to make code run faster --
125 * relies on fact that TYPE_i == i.
126 */
127#define MAX_TYPES 5 /* max number of types above */
128
129#define NSHUFF 50 /* to drop some "seed -> 1st value" linearity */

131static const int degrees[MAX_TYPES] = { DEG_0, DEG_1, DEG_2, DEG_3, DEG_4 };
132static const int seps [MAX_TYPES] = { SEP_0, SEP_1, SEP_2, SEP_3, SEP_4 };

134/*
135 * Initially, everything is set up as if from:
136 *
137 * initstate(1, randtbl, 128);
138 *
139 * Note that this initialization takes advantage of the fact that srandom()
140 * advances the front and rear pointers 10*rand_deg times, and hence the
141 * rear pointer which starts at 0 will also end up at zero; thus the zeroeth
142 * element of the state information, which contains info about the current
143 * position of the rear pointer is just
144 *
145 * MAX_TYPES * (rptr - state) + TYPE_3 == TYPE_3.
146 */
147
148static uint32_t randtbl[DEG_3 + 1] = {
149 TYPE_3,
150 0x991539b1, 0x16a5bce3, 0x6774a4cd, 0x3e01511e, 0x4e508aaa, 0x61048c05,
151 0xf5500617, 0x846b7115, 0x6a19892c, 0x896a97af, 0xdb48f936, 0x14898454,
152 0x37ffd106, 0xb58bff9c, 0x59e17104, 0xcf918a49, 0x09378c83, 0x52c7a471,
153 0x8d293ea9, 0x1f4fc301, 0xc3db71be, 0x39b44e1c, 0xf8a44ef9, 0x4c8b80b1,
154 0x19edc328, 0x87bf4bdd, 0xc9b240e5, 0xe9ee4b1b, 0x4382aee7, 0x535b6b41,
155 0xf3bec5da
156};

158/*
159 * fptr and rptr are two pointers into the state info, a front and a rear
160 * pointer. These two pointers are always rand_sep places apart, as they
161 * cycle cyclically through the state information. (Yes, this does mean we
162 * could get away with just one pointer, but the code for random() is more
163 * efficient this way). The pointers are left positioned as they would be
164 * from the call
165 *
166 * initstate(1, randtbl, 128);
167 *
168 * (The position of the rear pointer, rptr, is really 0 (as explained above
169 * in the initialization of randtbl) because the state table pointer is set
170 * to point to randtbl[1] (as explained below).
171 */
172static uint32_t *fptr = &randtbl[SEP_3 + 1];
173static uint32_t *rptr = &randtbl[1];
174
175/*
176 * The following things are the pointer to the state information table, the
177 * type of the current generator, the degree of the current polynomial being
178 * used, and the separation between the two pointers. Note that for efficiency
179 * of random(), we remember the first location of the state information, not
180 * the zeroeth. Hence it is valid to access state[-1], which is used to
181 * store the type of the R.N.G. Also, we remember the last location, since
182 * this is more efficient than indexing every time to find the address of
183 * the last element to see if the front and rear pointers have wrapped.
184 */
185static uint32_t *state = &randtbl[1];
186static int rand_type = TYPE_3;
187static int rand_deg = DEG_3;
188static int rand_sep = SEP_3;
189static uint32_t *end_ptr = &randtbl[DEG_3 + 1];
190
191static inline uint32_t good_rand(int32_t) __attribute__((always_inline));
192
193static inline uint32_t good_rand (int32_t x)
194{
195/*
196 * Compute x = (7^5 * x) mod (2^31 - 1)
197 * without overflowing 31 bits:
198 * (2^31 - 1) = 127773 * (7^5) + 2836
199 * From "Random number generators: good ones are hard to find",
200 * Park and Miller, Communications of the ACM, vol. 31, no. 10,
201 * October 1988, p. 1195.
202 */
203 int32_t hi, lo;
204
205 /* Can't be initialized with 0, so use another value. */
206 if (x == 0)
207 x = 123459876;
208 hi = x / 127773;
209 lo = x % 127773;
210 x = 16807 * lo - 2836 * hi;
211 if (x < 0)
212 x += 0x7fffffff;
213 return (x);
214}

216/*
217 * srandom:
218 *
219 * Initialize the random number generator based on the given seed. If the
220 * type is the trivial no-state-information type, just remember the seed.
221 * Otherwise, initializes state[] based on the given "seed" via a linear
222 * congruential generator. Then, the pointers are set to known locations
223 * that are exactly rand_sep places apart. Lastly, it cycles the state
224 * information a given number of times to get rid of any initial dependencies
225 * introduced by the L.C.R.N.G. Note that the initialization of randtbl[]
226 * for default usage relies on values produced by this routine.
227 */
228void
229bsd_srandom(unsigned x)
230{
231 int i, lim;
232
233 state[0] = (uint32_t)x;
234 if (rand_type == TYPE_0)
235 lim = NSHUFF;
236 else {
237 for (i = 1; i < rand_deg; i++)
238 state[i] = good_rand(state[i - 1]);
239 fptr = &state[rand_sep];
240 rptr = &state[0];
241 lim = 10 * rand_deg;
242 }
243 for (i = 0; i < lim; i++)
244 (void)bsd_random();
245}

247/*
248 * initstate:
249 *
250 * Initialize the state information in the given array of n bytes for future
251 * random number generation. Based on the number of bytes we are given, and
252 * the break values for the different R.N.G.'s, we choose the best (largest)
253 * one we can and set things up for it. srandom() is then called to
254 * initialize the state information.
255 *
256 * Note that on return from srandom(), we set state[-1] to be the type
257 * multiplexed with the current value of the rear pointer; this is so
258 * successive calls to initstate() won't lose this information and will be
259 * able to restart with setstate().
260 *
261 * Note: the first thing we do is save the current state, if any, just like
262 * setstate() so that it doesn't matter when initstate is called.
263 *
264 * Returns a pointer to the old state.
265 *
266 * Note: The Sparc platform requires that arg_state begin on an int
267 * word boundary; otherwise a bus error will occur. Even so, lint will
268 * complain about mis-alignment, but you should disregard these messages.
269 */
270char *
271bsd_initstate(
272 unsigned seed, /* seed for R.N.G. */
273 char *arg_state, /* pointer to state array */
274 size_t n /* # bytes of state info */
275) {
276 char *ostate = (char *)(&state[-1]);
277 uint32_t *int_arg_state = (uint32_t *)arg_state;
278
279 if (rand_type == TYPE_0)
280 state[-1] = rand_type;
281 else
282 state[-1] = MAX_TYPES * (rptr - state) + rand_type;
283 if (n < BREAK_0) {
284 (void)fprintf(stderr,
285 "random: not enough state (%zd bytes); ignored.\n", n);
286 return(0);
287 }
288 if (n < BREAK_1) {
289 rand_type = TYPE_0;
290 rand_deg = DEG_0;
291 rand_sep = SEP_0;
292 } else if (n < BREAK_2) {
293 rand_type = TYPE_1;
294 rand_deg = DEG_1;
295 rand_sep = SEP_1;
296 } else if (n < BREAK_3) {
297 rand_type = TYPE_2;
298 rand_deg = DEG_2;
299 rand_sep = SEP_2;
300 } else if (n < BREAK_4) {
301 rand_type = TYPE_3;
302 rand_deg = DEG_3;
303 rand_sep = SEP_3;
304 } else {
305 rand_type = TYPE_4;
306 rand_deg = DEG_4;
307 rand_sep = SEP_4;
308 }
309 state = int_arg_state + 1; /* first location */
310 end_ptr = &state[rand_deg]; /* must set end_ptr before srandom */
311 bsd_srandom(seed);
312 if (rand_type == TYPE_0)
313 int_arg_state[0] = rand_type;
314 else
315 int_arg_state[0] = MAX_TYPES * (rptr - state) + rand_type;
316 return(ostate);
317}

319/*
320 * setstate:
321 *
322 * Restore the state from the given state array.
323 *
324 * Note: it is important that we also remember the locations of the pointers
325 * in the current state information, and restore the locations of the pointers
326 * from the old state information. This is done by multiplexing the pointer
327 * location into the zeroeth word of the state information.
328 *
329 * Note that due to the order in which things are done, it is OK to call
330 * setstate() with the same state as the current state.
331 *
332 * Returns a pointer to the old state information.
333 *
334 * Note: The Sparc platform requires that arg_state begin on an int
335 * word boundary; otherwise a bus error will occur. Even so, lint will
336 * complain about mis-alignment, but you should disregard these messages.
337 */
338char *
339bsd_setstate(
340 const char *arg_state /* pointer to state array */
341) {
342 uint32_t *new_state = (uint32_t *)arg_state;
343 uint32_t type = new_state[0] % MAX_TYPES;
344 uint32_t rear = new_state[0] / MAX_TYPES;
345 char *ostate = (char *)(&state[-1]);
346
347 if (rand_type == TYPE_0)
348 state[-1] = rand_type;
349 else
350 state[-1] = MAX_TYPES * (rptr - state) + rand_type;
351 switch(type) {
352 case TYPE_0:
353 case TYPE_1:
354 case TYPE_2:
355 case TYPE_3:
356 case TYPE_4:
357 rand_type = type;
358 rand_deg = degrees[type];
359 rand_sep = seps[type];
360 break;
361 default:
362 (void)fprintf(stderr,
363 "random: state info corrupted; not changed.\n");
364 }
365 state = new_state + 1;
366 if (rand_type != TYPE_0) {
367 rptr = &state[rear];
368 fptr = &state[(rear + rand_sep) % rand_deg];
369 }
370 end_ptr = &state[rand_deg]; /* set end_ptr too */
371 return(ostate);
372}

374/*
375 * random:
376 *
377 * If we are using the trivial TYPE_0 R.N.G., just do the old linear
378 * congruential bit. Otherwise, we do our fancy trinomial stuff, which is
379 * the same in all the other cases due to all the global variables that have
380 * been set up. The basic operation is to add the number at the rear pointer
381 * into the one at the front pointer. Then both pointers are advanced to
382 * the next location cyclically in the table. The value returned is the sum
383 * generated, reduced to 31 bits by throwing away the "least random" low bit.
384 *
385 * Note: the code takes advantage of the fact that both the front and
386 * rear pointers can't wrap on the same call by not testing the rear
387 * pointer if the front one has wrapped.
388 *
389 * Returns a 31-bit random number.
390 */
391long
392bsd_random(void)
393{
394 uint32_t i;
395 uint32_t *f, *r;
396
397 if (rand_type == TYPE_0) {
398 i = state[0];
399 state[0] = i = (good_rand(i)) & 0x7fffffff;
400 } else {
401 /*
402 * Use local variables rather than static variables for speed.
403 */
404 f = fptr; r = rptr;
405 *f += *r;
406 i = (*f >> 1) & 0x7fffffff; /* chucking least random bit */
407 if (++f >= end_ptr) {
408 f = state;
409 ++r;
410 }
411 else if (++r >= end_ptr) {
412 r = state;
413 }
414
415 fptr = f; rptr = r;
416 }
417 return((long)i);
418}
