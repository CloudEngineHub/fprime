90void configureTopology() {
91 // Command sequencer needs to allocate memory to hold contents of command sequences
92 cmdSeq.allocateBuffer(0, mallocator, CMD_SEQ_BUFFER_SIZE);
93
94 // Rate group driver needs a divisor list
95 rateGroupDriverComp.configure(rateGroupDivisorsSet);
96
97 // Rate groups require context arrays. Empty for Reference example.
98 rateGroup1Comp.configure(rateGroup1Context, FW_NUM_ARRAY_ELEMENTS(rateGroup1Context));
99 rateGroup2Comp.configure(rateGroup2Context, FW_NUM_ARRAY_ELEMENTS(rateGroup2Context));
100 rateGroup3Comp.configure(rateGroup3Context, FW_NUM_ARRAY_ELEMENTS(rateGroup3Context));
101
102 // File downlink requires some project-derived properties.
103 fileDownlink.configure(FILE_DOWNLINK_TIMEOUT, FILE_DOWNLINK_COOLDOWN, FILE_DOWNLINK_CYCLE_TIME,
104 FILE_DOWNLINK_FILE_QUEUE_DEPTH);
105
106 // Parameter database is configured with a database file name, and that file must be initially read.
107 prmDb.configure("PrmDb.dat");
108 prmDb.readParamFile();
109
110 // Health is supplied a set of ping entires.
111 health.setPingEntries(pingEntries, FW_NUM_ARRAY_ELEMENTS(pingEntries), HEALTH_WATCHDOG_CODE);
112
113 // Buffer managers need a configured set of buckets and an allocator used to allocate memory for those buckets.
114 Svc::BufferManager::BufferBins upBuffMgrBins;
115 memset(&upBuffMgrBins, 0, sizeof(upBuffMgrBins));
116 upBuffMgrBins.bins[0].bufferSize = UPLINK_BUFFER_MANAGER_STORE_SIZE;
117 upBuffMgrBins.bins[0].numBuffers = UPLINK_BUFFER_MANAGER_QUEUE_SIZE;
118 fileUplinkBufferManager.setup(UPLINK_BUFFER_MANAGER_ID, 0, mallocator, upBuffMgrBins);
119
120 Svc::BufferManager::BufferBins dpBuffMgrBins;
121 memset(&dpBuffMgrBins, 0, sizeof(dpBuffMgrBins));
122 dpBuffMgrBins.bins[0].bufferSize = DP_BUFFER_MANAGER_STORE_SIZE;
123 dpBuffMgrBins.bins[0].numBuffers = DP_BUFFER_MANAGER_QUEUE_SIZE;
124 dpBufferManager.setup(DP_BUFFER_MANAGER_ID, 0, mallocator, dpBuffMgrBins);
125
126 // Framer and Deframer components need to be passed a protocol handler
127 downlink.setup(framing);
128 uplink.setup(deframing);
129
130 Fw::FileNameString dpDir("./DpCat");
131
132 // create the DP directory if it doesn't exist
133 Os::FileSystem::createDirectory(dpDir.toChar());
134
135 dpCat.configure(&dpDir,1,0,mallocator);
136 dpWriter.configure(dpDir);
137
138 // Note: Uncomment when using Svc:TlmPacketizer
139 //tlmSend.setPacketList(RefPacketsPkts, RefPacketsIgnore, 1);
140}

144void setupTopology(const TopologyState& state) {
145 // Autocoded initialization. Function provided by autocoder.
146 initComponents(state);
147 // Autocoded id setup. Function provided by autocoder.
148 setBaseIds();
149 // Autocoded connection wiring. Function provided by autocoder.
150 connectComponents();
151 // Autocoded command registration. Function provided by autocoder.
152 regCommands();
153 // Autocoded configuration. Function provided by autocoder.
154 configComponents(state);
155 // Project-specific component configuration. Function provided above. May be inlined, if desired.
156 configureTopology();
157 // Autocoded parameter loading. Function provided by autocoder.
158 loadParameters();
159 // Autocoded task kick-off (active components). Function provided by autocoder.
160 startTasks(state);
161 // Startup TLM and Config verbosity for Versions
162 version.config(true);
163 // Initialize socket client communication if and only if there is a valid specification
164 if (state.hostname != nullptr && state.port != 0) {
165 Os::TaskString name("ReceiveTask");
166 // Uplink is configured for receive so a socket task is started
167 comm.configure(state.hostname, state.port);
168 comm.start(name, true, COMM_PRIORITY, Default::STACK_SIZE);
169 }
170}

172// Variables used for cycle simulation
173Os::Mutex cycleLock;
174volatile bool cycleFlag = true;
175
176void startSimulatedCycle(Fw::Time interval) {
177 cycleLock.lock();
178 bool cycling = cycleFlag;
179 cycleLock.unLock();
180
181 // Main loop
182 while (cycling) {
183 Ref::blockDrv.callIsr();
184 Os::Task::delay(interval);
185
186 cycleLock.lock();
187 cycling = cycleFlag;
188 cycleLock.unLock();
189 }
190}
191
192void stopSimulatedCycle() {
193 cycleLock.lock();
194 cycleFlag = false;
195 cycleLock.unLock();
196}

210}
