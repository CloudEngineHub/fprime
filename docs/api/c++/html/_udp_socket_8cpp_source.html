87SocketIpStatus UdpSocket::bind(NATIVE_INT_TYPE fd) {
88 struct sockaddr_in address;
89 FW_ASSERT(-1 != fd);
90
91 // Set up the address port and name
92 address.sin_family = AF_INET;
93 this->m_lock.lock();
94 address.sin_port = htons(m_recv_port);
95 this->m_lock.unlock();
96 // OS specific settings
97#if defined TGT_OS_TYPE_VXWORKS || TGT_OS_TYPE_DARWIN
98 address.sin_len = static_cast<U8>(sizeof(struct sockaddr_in));
99#endif
100
101 // First IP address to socket sin_addr
102 if (IpSocket::addressToIp4(m_recv_hostname, &address.sin_addr) != SOCK_SUCCESS) {
103 return SOCK_INVALID_IP_ADDRESS;
104 };
105 // UDP (for receiving) requires bind to an address to the socket
106 if (::bind(fd, reinterpret_cast<struct sockaddr*>(&address), sizeof(address)) < 0) {
107 return SOCK_FAILED_TO_BIND;
108 }
109
110 socklen_t size = sizeof(address);
111 if (::getsockname(fd, reinterpret_cast<struct sockaddr *>(&address), &size) == -1) {
112 return SOCK_FAILED_TO_READ_BACK_PORT;
113 }
114 U16 port = ntohs(address.sin_port);
115
116 this->m_lock.lock();
117 FW_ASSERT(sizeof(this->m_state->m_addr_recv) == sizeof(address), sizeof(this->m_state->m_addr_recv), sizeof(address));
118 memcpy(&this->m_state->m_addr_recv, &address, sizeof(this->m_state->m_addr_recv));
119 this->m_recv_port = port;
120 this->m_lock.unlock();
121
122 return SOCK_SUCCESS;
123}

125SocketIpStatus UdpSocket::openProtocol(NATIVE_INT_TYPE& fd) {
126 SocketIpStatus status = SOCK_SUCCESS;
127 NATIVE_INT_TYPE socketFd = -1;
128 struct sockaddr_in address;
129
130 this->m_lock.lock();
131 U16 port = this->m_port;
132 this->m_lock.unlock();
133
134 // Acquire a socket, or return error
135 if ((socketFd = ::socket(AF_INET, SOCK_DGRAM, 0)) == -1) {
136 return SOCK_FAILED_TO_GET_SOCKET;
137 }
138
139 // May not be sending in all cases
140 if (port != 0) {
141 // Set up the address port and name
142 address.sin_family = AF_INET;
143 this->m_lock.lock();
144 address.sin_port = htons(this->m_port);
145 this->m_lock.unlock();
146
147 // OS specific settings
148#if defined TGT_OS_TYPE_VXWORKS || TGT_OS_TYPE_DARWIN
149 address.sin_len = static_cast<U8>(sizeof(struct sockaddr_in));
150#endif
151
152 // First IP address to socket sin_addr
153 if ((status = IpSocket::addressToIp4(m_hostname, &(address.sin_addr))) != SOCK_SUCCESS) {
154 ::close(socketFd);
155 return status;
156 };
157
158 // Now apply timeouts
159 if ((status = IpSocket::setupTimeouts(socketFd)) != SOCK_SUCCESS) {
160 ::close(socketFd);
161 return status;
162 }
163 this->m_lock.lock();
164 FW_ASSERT(sizeof(this->m_state->m_addr_send) == sizeof(address), sizeof(this->m_state->m_addr_send),
165 sizeof(address));
166 memcpy(&this->m_state->m_addr_send, &address, sizeof(this->m_state->m_addr_send));
167 this->m_lock.unlock();
168 }
169
170 // When we are setting up for receiving as well, then we must bind to a port
171 if ((status = this->bind(socketFd)) != SOCK_SUCCESS) {
172 ::close(socketFd);
173 return status; // Not closing FD as it is still a valid send FD
174 }
175 this->m_lock.lock();
176 U16 recv_port = this->m_recv_port;
177 this->m_lock.unlock();
178 // Log message for UDP
179 if (port == 0) {
180 Fw::Logger::log("Setup to receive udp at %s:%hu\n", m_recv_hostname,
181 recv_port);
182 } else {
183 Fw::Logger::log("Setup to receive udp at %s:%hu and send to %s:%hu\n",
184 m_recv_hostname,
185 recv_port,
186 m_hostname,
187 port);
188 }
189 FW_ASSERT(status == SOCK_SUCCESS, status);
190 fd = socketFd;
191 return status;
192}

194I32 UdpSocket::sendProtocol(const U8* const data, const U32 size) {
195 FW_ASSERT(this->m_state->m_addr_send.sin_family != 0); // Make sure the address was previously setup
196 return static_cast<I32>(::sendto(this->m_fd, data, size, SOCKET_IP_SEND_FLAGS,
197 reinterpret_cast<struct sockaddr *>(&this->m_state->m_addr_send), sizeof(this->m_state->m_addr_send)));
198}

200I32 UdpSocket::recvProtocol(U8* const data, const U32 size) {
201 FW_ASSERT(this->m_state->m_addr_recv.sin_family != 0); // Make sure the address was previously setup
202 return static_cast<I32>(::recvfrom(this->m_fd, data, size, SOCKET_IP_RECV_FLAGS, nullptr, nullptr));
203}
