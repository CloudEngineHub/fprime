97 ) {
98 assert(lower <= upper);
99 const F64 length = static_cast<F64>(upper) - lower + 1;
100 const F64 randFloat = inUnitInterval() * length;
101 const U32 offset = static_cast<U32>(randFloat);
102 const U32 result = lower + offset;
103 // Handle boundary case where inUnitInterval returns 1.0
104 return (result <= upper) ? result : result - 1;
