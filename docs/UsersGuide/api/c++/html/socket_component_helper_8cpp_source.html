91 SocketIpStatus SocketComponentHelper::send(const U8* const data, const U32 size) {
92  SocketIpStatus status = SOCK_SUCCESS;
93  this->m_lock.lock();
94  NATIVE_INT_TYPE fd = this->m_fd;
95  this->m_lock.unlock();
96  // Prevent transmission before connection, or after a disconnect
97  if (fd == -1) {

159 void SocketComponentHelper::readTask(void* pointer) {
160  FW_ASSERT(pointer);
161  SocketIpStatus status = SOCK_SUCCESS;
162  SocketComponentHelper* self = reinterpret_cast<SocketComponentHelper*>(pointer);
163  do {
164  // Prevent transmission before connection, or after a disconnect
165  if ((not self->isOpened()) and (not self->m_stop)) {
166  status = self->reconnect();
167  if(status != SOCK_SUCCESS) {
168  Fw::Logger::log(
169  "[WARNING] Failed to open port with status %d and errno %d\n",
170  status,
171  errno);
172  (void) Os::Task::delay(SOCKET_RETRY_INTERVAL);
173  continue;
174  }
175  }
176  // If the network connection is open, read from it
177  if (self->isStarted() and self->isOpened() and (not self->m_stop)) {
178  Fw::Buffer buffer = self->getBuffer();
179  U8* data = buffer.getData();
180  FW_ASSERT(data);
181  U32 size = buffer.getSize();
182  // recv blocks, so it may have been a while since its done an isOpened check
183  status = self->recv(data, size);
184  if ((status != SOCK_SUCCESS) && (status != SOCK_INTERRUPTED_TRY_AGAIN && (status != SOCK_NO_DATA_AVAILABLE))) {
185  Fw::Logger::log("[WARNING] Failed to recv from port with status %d and errno %d\n",
186  status,
187  errno);
188  self->close();
189  buffer.setSize(0);
190  } else {
191  // Send out received data
192  buffer.setSize(size);
193  }
194  self->sendBuffer(buffer, status);
195  }
196  }
197  // As long as not told to stop, and we are successful interrupted or ordered to retry, keep receiving
198  while (not self->m_stop &&
199  (status == SOCK_SUCCESS || status == SOCK_INTERRUPTED_TRY_AGAIN || self->m_reconnect));
200  self->shutdown(); // Shutdown the port entirely
201 }
